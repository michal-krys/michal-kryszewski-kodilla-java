SELECT I.SUMMARY, I.DESCRIPTION, L.LISTNAME
FROM ISSUESLIST L JOIN ISSUES I ON L.ID = I.ISSUESLIST_ID;

SELECT I.SUMMARY, I.DESCRIPTION, U.FIRSTNAME, U.LASTNAME
FROM USERS U JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TASKS_ASSIGNED
FROM USERS U JOIN ISSUES I on U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY I.USER_ID_ASSIGNEDTO
HAVING COUNT(*) > 1;
